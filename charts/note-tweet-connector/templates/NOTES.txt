1. Get the webhook endpoint URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  Webhook URL: http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "chart.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Webhook URL: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "chart.fullname" . }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "chart.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Webhook URL: http://$SERVICE_IP:{{ .Values.service.port }}"
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "chart.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Webhook port-forward: kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:{{ .Values.args.port }}"
{{- end }}

2. Health check endpoint: /healthz

3. Metrics endpoint:
{{- if .Values.metricsService.enabled }}
  Metrics are exposed on a separate service: {{ include "chart.fullname" . }}-metrics:{{ .Values.metricsService.port }}
  Port-forward: kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "chart.fullname" . }}-metrics {{ .Values.metricsService.port }}:{{ .Values.metricsService.port }}
{{- else }}
  Metrics are available at /metrics on port {{ .Values.args.metricsPort }}
{{- end }}
